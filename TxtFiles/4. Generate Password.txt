<!DOCTYPE html>
<html>
  <head>
    <title>Generate Password</title>
    <script>
      function generatePassword(length) {
        // Define possible characters
        const chars = "ABCDEFGHIJKLMNOPQRSTUVWXYZabcdefghijklmnopqrstuvwxyz0123456789!@#$%^&*()_+";

        let password = "";
        for (let i = 0; i < length; i++) {
          // Add a random character to the password
          password += chars.charAt(Math.floor(Math.random() * chars.length));
        }

        return password;
      }

      function onGenerateClick() {
        // Get the length from the input field
        const length = parseInt(document.getElementById("passwordLength").value);

        // Generate the password
        const password = generatePassword(length);

        // Set the password value to the output field
        document.getElementById("passwordOutput").value = password;
      }
    </script>
  </head>
  <body>
    <label for="passwordLength">Password Length:</label>
    <input type="number" id="passwordLength" min="1" max="50" value="8">
    <br>
    <br>
    <button onclick="onGenerateClick()">Generate Password</button>
    <br>
    <br>
    <label for="passwordOutput">Generated Password:</label>
    <input type="text" id="passwordOutput" readonly>
  </body>
</html>
